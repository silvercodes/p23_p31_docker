```bash
# Показать запущенные контейнеры
docker ps

# Показать все контейнеры
docker ps -a

# Показать имеющиеся образы в локальном хранилище
docker images

# Остановить контейнер по имени или по ID
docker stop <container_id>

# Запустить остановленный контейнер по имени или по ID
docker start <container_id>

# удалить остановленный контейнер
docker rm <container_id>

# Удалить образ (свободный от контейнеров)
docker rmi <image_id>

# собрать образ из Dockerfile в текущей директории (`.`). Флаг `-t` задает имя тега.
docker build -t <image_tag> .

# Вывести логи контейнера
docker logs <container_id>

# Выводить логи контейнера в реальном времени
docker logs -f <container_id>

# Вывести 100 последних строк логов
docker logs --tail 100 <container_id>
```
